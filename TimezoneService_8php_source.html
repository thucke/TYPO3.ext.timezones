185  public function setTimezoneArray(): void
186  {
187  $timezone_identifiers_list = DateTimeZone::listIdentifiers();
188  foreach ($timezone_identifiers_list as $timezone_identifier) {
189  $dateTimeZone = new DateTimeZone($timezone_identifier);
190  $dateTime = new DateTime('now', $dateTimeZone);
191  //$abbreviation = $this->intlDateFormatter::formatObject($dateTime,'zzzz', $this->locale);
192  $hours = floor($dateTimeZone->getOffset($dateTime) / 3600);
193  $mins = floor(($dateTimeZone->getOffset($dateTime) - ($hours * 3600)) / 60);
194  $hours = 'GMT'.($hours < 0 ? $hours : '+'.$hours);
195  $mins = ($mins > 0 ? $mins : '0'.$mins);
196  $text = str_replace('_', ' ', $timezone_identifier);
197  $this->timezoneArray[$timezone_identifier] = $text.' ('.$hours.':'.$mins.')';
198  }
199  }
