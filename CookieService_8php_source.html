41 $cookieDomain = $GLOBALS['TYPO3_CONF_VARS']['FE']['cookieDomain'];
42 }
43 if ($cookieDomain) {
44 if ($cookieDomain[0] === '/') {
45 $match = [];
46 $matchCnt = @preg_match($cookieDomain, GeneralUtility::getIndpEnv('TYPO3_HOST_ONLY'), $match);
